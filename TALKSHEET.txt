import java.io.*;
import java.lang.*;

public class Main
{
	public static void main(String[] args) throws Exception {
	    BufferedReader dataIn = new BufferedReader (new InputStreamReader(System.in));
	    
	    String input = " ";
	    double a=0, b=0, c=0, Side=0,area=0;

	    System.out.println("Input the length of 1st side of the triangle: ");
	    input = dataIn.readLine();
	    a = Double.parseDouble(input);
	    
	    System.out.println("Input the length of 2nd side of the triangle: ");
	    input = dataIn.readLine();
	    b = Double.parseDouble(input);
	    
	    System.out.println("Input the length of 3rd side of the triangle: ");
	    input = dataIn.readLine();
	    c = Double.parseDouble(input);
	    
	    Side = (a+b+c)/2;
	    area = Math.sqrt(Side*(Side-a)*(Side-b)*(Side-c));
	    
        String formattedArea = String.format("%.4f", area);
	    System.out.println("The area of the triangle is: "+area);
	}
}

	    String input = " ";
	    double rad=0, circum=0,area=0,PI=3.14;

	    System.out.println("Input the radius(1/2 of diameter) of a circle: ");
	    input = dataIn.readLine();
	    rad = Double.parseDouble(input);
	    
	    
	    circum = 2*PI*rad;
	    area = PI*(rad*rad);

        String formattedArea = String.format("%.4f", area, circum);
	    System.out.println("The area of the circle is: "+area);
	    System.out.println("The circumference of the circle is:  "+circum);
	}
}

{

	    f = 9.0/5.0*(k-273.15) + 32;

	}
}
